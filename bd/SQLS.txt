
CREATE TABLE Dep (
Nome varchar(100),
CodDep VARCHAR(3) PRIMARY KEY 
)


CREATE TABLE User (
Nome varchar(100),
Email varchar(100),
Senha varchar(100),
Tipo varchar(100),
CodUsuarios INTEGER PRIMARY KEY   NOT NULL GENERATED ALWAYS AS IDENTITY ,
CodDep VARCHAR(3),
FOREIGN KEY(CodDep) REFERENCES Dep (CodDep)
)
