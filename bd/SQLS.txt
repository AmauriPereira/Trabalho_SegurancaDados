
CREATE TABLE Departamento (
Nome varchar(100),
Cod_Departamento VARCHAR(3) PRIMARY KEY 
)


CREATE TABLE USUARIO(
Nome varchar(100) NOT NULL,
Email varchar(50) NOT NULL,
Senha varchar(10) NOT NULL,
Tipo varchar(100) NOT NULL,
Id_Usuario INTEGER PRIMARY KEY   NOT NULL GENERATED ALWAYS AS IDENTITY ,
Cod_Departamento VARCHAR(3),
FOREIGN KEY(Cod_Departamento ) REFERENCES Departamento (Cod_Departamento)
)

CREATE TABLE PROJETO(
Id_PROJETO INTEGER PRIMARY KEY   NOT NULL GENERATED ALWAYS AS IDENTITY ,
Nome varchar(100) NOT NULL,
DESCRICAO varchar(200) NOT NULL,
DATA_INCIO DATE NOT NULL,
DATA_TERMINO DATE NOT NULL,
Cod_Departamento VARCHAR(3),
FOREIGN KEY(Cod_Departamento ) REFERENCES Departamento (Cod_Departamento)
)

CREATE TABLE ATIVIDADE(
Id_ATIVIDADE INTEGER PRIMARY KEY   NOT NULL GENERATED ALWAYS AS IDENTITY ,
Nome varchar(100) NOT NULL,
Duracao Datetime(200) NOT NULL,
ID_PROJETO  int,
ID_USUARIO int,
FOREIGN KEY (ID_PROJETO) REFERENCES Departamento (ID_PROJETO),
FOREIGN KEY(ID_USUARIO) REFERENCES Departamento (ID_USUARIO)
)
